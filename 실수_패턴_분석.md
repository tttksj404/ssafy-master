def melt_once(grid):
    q = deque([(0, 0)])
    visited = [[False] * M for _ in range(N)]
    visited[0][0] = True
    to_melt = []

    while q:
        r, c = q.popleft()
        for d in range(4):
            nr, nc = r + dr[d], c + dc[d]
            if 0 <= nr < N and 0 <= nc < M and not visited[nr][nc]:
                visited[nr][nc] = True
                if grid[nr][nc] == 0:
                    q.append((nr, nc))
                else:  # grid[nr][nc] == 1
                    to_melt.append((nr, nc))

    for r, c in to_melt:
        grid[r][c] = 0
    return len(to_melt)
